def init(dims):
    np.random.seed(3)
    parameters = {}
    n = len(dims)
    for i in range(1,n):
        parameters['W'+str(i)] = np.random.randn(dims[i],dims[i-1])
        parameters['b'+str(i)] = np.zeros([dims[i],1])
    return parameters

def forward_propagation(data,parameters):
    n = len(parameters)//2
    ZA = {}
    a = data
    for i in range(0,n):
        z = np.dot(parameters['W'+str(i+1)],a)+parameters['b'+str(i+1)]
        ZA['z'+str(i+1)] = a
        a = np.tanh(z)
        ZA['a'+str(i+1)] = a
    return a,ZA

def lossFunction(dataY,a,parameters,lamda=0.005):
    m = dataY.shape[1]
    #print(m)
    n = len(parameters)//2
    W = 0.0
    for i in range(0,n):
        W += (parameters['W'+str(i+1)]**2).sum()
    J = (np.power((dataY-a),2).sum()/m + lamda * W)/2 
    #J = (np.power((dataY-a),2).sum()/m)/2 
    return J

def back_propation(data,dataY,a,ZA,parameters,alpha=0.05): #autoEncoderdatadataY alpha:
    m = dataY.shape[1]
    dparams = {}
    n = len(parameters)//2
    dZn = a - dataY
    dWn = np.dot(dZn,ZA['a'+str(n-1)].T)/m
    dbn = np.sum(dZn,axis = 1, keepdims = True)/m
    dparams['dZ'+str(n)] = dZn
    dparams['dW'+str(n)] = dWn
    dparams['db'+str(n)] = dbn
    parameters['W'+str(n)] = parameters['W'+str(n)] - alpha*dWn
    parameters['b'+str(n)] = parameters['b'+str(n)] - alpha*dbn
    for i in reversed(range(n-1)): #1,0
        dZi = np.multiply(np.dot(parameters["W"+str(i+1+1)].T,dparams["dZ"+str(i+1+1)]),(1 - np.power(ZA['a'+str(i+1)], 2)))
        if i != 0:
            dWi = np.dot(dZi,ZA['a'+str(i)].T)/m
        else:
            dWi = np.dot(dZi,data.T)/m
        dbi = np.sum(dZi,axis = 1, keepdims = True)/m
        dparams['dZ'+str(i+1)] = dZi
        dparams['dW'+str(i+1)] = dWi
        dparams['db'+str(i+1)] = dbi
        parameters['W'+str(i+1)] = parameters['W'+str(i+1)] - alpha*dWi
        parameters['b'+str(i+1)] = parameters['b'+str(i+1)] - alpha*dbi
    return parameters,dparams
